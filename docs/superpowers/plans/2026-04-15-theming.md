# Theming Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add server-side theme packs (folder + `theme.json`) that style the terminal frame, toolbar slots, and interactive controls in tmux-web, with hot-switching via the menu and a single global font picker populated from all installed packs.

**Architecture:** Theme packs live at top-level `themes/` (bundled into binary) or `~/.config/tmux-web/themes/` (user dir). Server scans both, exposes `/api/themes`, `/api/fonts`, and `/themes/<pack>/<file>`. Client adds frame divs + toolbar slot wrappers, loads all fonts at startup, swaps a `<link id="theme-css">` tag to switch themes. Current chrome ports to `themes/default/` so there is no "no-theme" code path.

**Tech Stack:** Bun runtime, TypeScript, vanilla DOM, `bun test`, Playwright. Spec: `docs/superpowers/specs/2026-04-15-theming-design.md`.

```ts
import fs from 'fs';
import path from 'path';

export type FontInfo  = { family: string; file: string; pack: string; packDir: string };
export type ThemeInfo = {
  name: string; pack: string; css: string;
  defaultFont?: string; author?: string; version?: string;
  source: 'user' | 'bundled';
};
export type PackInfo = {
  dir: string;            // folder name (used in URL)
  fullPath: string;       // absolute path on disk
  source: 'user' | 'bundled';
  manifest: PackManifest;
};
export type PackManifest = {
  author?: string;
  version?: string;
  fonts?: { file: string; family: string }[];
  themes?: { name: string; css: string; defaultFont?: string }[];
};

function readManifest(packPath: string): PackManifest | null {
  const file = path.join(packPath, 'theme.json');
  try {
    const raw = fs.readFileSync(file, 'utf8');
    const m = JSON.parse(raw);
    if (typeof m !== 'object' || m === null) return null;
    return m as PackManifest;
  } catch {
    return null;
  }
}

function scanDir(dir: string, source: 'user' | 'bundled'): PackInfo[] {
  if (!fs.existsSync(dir)) return [];
  const out: PackInfo[] = [];
  for (const entry of fs.readdirSync(dir)) {
    const full = path.join(dir, entry);
    let stat;
    try { stat = fs.statSync(full); } catch { continue; }
    if (!stat.isDirectory()) continue;
    const m = readManifest(full);
    if (!m) {
      console.warn(`[themes] skipping pack '${entry}' in ${dir}: invalid theme.json`);
      continue;
    }
    out.push({ dir: entry, fullPath: full, source, manifest: m });
  }
  return out;
}

export function listPacks(userDir: string | null, bundledDir: string | null): PackInfo[] {
  const user = userDir ? scanDir(userDir, 'user') : [];
  const bundled = bundledDir ? scanDir(bundledDir, 'bundled') : [];
  return [...user, ...bundled];
}

export function listThemes(packs: PackInfo[]): ThemeInfo[] {
  const seen = new Map<string, ThemeInfo>();
  for (const p of packs) {
    for (const t of p.manifest.themes ?? []) {
      if (!t.name || !t.css) continue;
      if (seen.has(t.name)) {
        console.warn(`[themes] duplicate theme name '${t.name}' in pack '${p.dir}' (${p.source}); ignoring`);
        continue;
      }
      seen.set(t.name, {
        name: t.name, pack: p.dir, css: t.css, defaultFont: t.defaultFont,
        author: p.manifest.author, version: p.manifest.version, source: p.source,
      });
    }
  }
  return [...seen.values()].sort((a, b) => a.name.localeCompare(b.name));
}

export function listFonts(packs: PackInfo[]): FontInfo[] {
  const seen = new Map<string, FontInfo>();
  for (const p of packs) {
    for (const f of p.manifest.fonts ?? []) {
      if (!f.family || !f.file) continue;
      if (seen.has(f.family)) continue;  // first wins (user dir scanned first)
      seen.set(f.family, { family: f.family, file: f.file, pack: p.dir, packDir: p.fullPath });
    }
  }
  return [...seen.values()].sort((a, b) => a.family.localeCompare(b.family));
}

export function resolveTheme(name: string, packs: PackInfo[]): ThemeInfo | null {
  return listThemes(packs).find(t => t.name === name) ?? null;
}

function findPack(packDir: string, packs: PackInfo[]): PackInfo | null {
  return packs.find(p => p.dir === packDir) ?? null;
}

export function readPackFile(packDir: string, file: string, packs: PackInfo[]): { fullPath: string } | null {
  if (!file || file.includes('..') || file.includes('/') || file.includes('\\') || file.startsWith('.')) {
    return null;
  }
  const pack = findPack(packDir, packs);
  if (!pack) return null;
  const full = path.join(pack.fullPath, file);
  if (!fs.existsSync(full)) return null;
  return { fullPath: full };
}
```

```ts
import os from 'os';
async function materializeBundledThemes(): Promise<string | null> {
  const keys = Object.keys(embeddedAssets).filter(k => k.startsWith('themes/'));
  if (keys.length === 0) return null;
  const root = path.join(os.tmpdir(), `tmux-web-themes-${process.pid}`);
  for (const key of keys) {
    const src = embeddedAssets[key]!;
    const dest = path.join(root, key.slice('themes/'.length));
    fs.mkdirSync(path.dirname(dest), { recursive: true });
    const bytes = new Uint8Array(await Bun.file(src).arrayBuffer());
    fs.writeFileSync(dest, bytes);
  }
  process.on('exit', () => { try { fs.rmSync(root, { recursive: true, force: true }); } catch {} });
  return root;
}
```

```ts
export type ThemeInfo = { name: string; pack: string; css: string;
  defaultFont?: string; author?: string; version?: string; source: 'user' | 'bundled' };
export type FontInfo  = { family: string; file: string; pack: string };
export type Insets = { top: number; right: number; bottom: number; left: number };

let cachedThemes: ThemeInfo[] | null = null;
let cachedFonts:  FontInfo[]  | null = null;
let activeTheme = 'Default';

export async function listThemes(): Promise<ThemeInfo[]> {
  if (!cachedThemes) {
    const r = await fetch('/api/themes');
    cachedThemes = r.ok ? await r.json() : [];
  }
  return cachedThemes!;
}

export async function listFonts(): Promise<FontInfo[]> {
  if (!cachedFonts) {
    const r = await fetch('/api/fonts');
    cachedFonts = r.ok ? await r.json() : [];
  }
  return cachedFonts!;
}

export function getActiveTheme(): string { return activeTheme; }

export async function loadAllFonts(): Promise<void> {
  const fonts = await listFonts();
  for (const f of fonts) {
    try {
      const ff = new FontFace(f.family, `url(/themes/${encodeURIComponent(f.pack)}/${encodeURIComponent(f.file)})`);
      await ff.load();
      (document as any).fonts.add(ff);
    } catch (e) {
      console.warn(`[theme] failed to load font ${f.family}:`, e);
    }
  }
}

export async function applyTheme(name: string): Promise<void> {
  const themes = await listThemes();
  let t = themes.find(x => x.name === name);
  if (!t) {
    console.error(`[theme] theme "${name}" not found, falling back to Default`);
    t = themes.find(x => x.name === 'Default');
    if (!t) return;
    name = 'Default';
  }
  const old = document.getElementById('theme-css');
  if (old) old.remove();
  const link = document.createElement('link') as any;
  link.id = 'theme-css';
  link.rel = 'stylesheet';
  link.href = `/themes/${encodeURIComponent(t.pack)}/${encodeURIComponent(t.css)}`;
  document.head.appendChild(link);
  activeTheme = name;
}

export function readBorderInsets(): Insets {
  const s = getComputedStyle(document.documentElement);
  const px = (k: string) => parseFloat(s.getPropertyValue(k)) || 0;
  return {
    top:    px('--tw-border-top'),
    right:  px('--tw-border-right'),
    bottom: px('--tw-border-bottom'),
    left:   px('--tw-border-left'),
  };
}

export function clearCaches(): void { cachedThemes = null; cachedFonts = null; }
```
